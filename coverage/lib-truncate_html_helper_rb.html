23   def truncate_html(input, *args)
 
24     # support both 2.2 & earlier APIs

31       encoder = HTMLEntities.new(TruncateHtmlHelper.flavor)
 
32       tags, output, chars_remaining = [], '', length
 
33

35         element = parser.pull
 
36         case element.event_type
 
37         when :start_element

48       end
